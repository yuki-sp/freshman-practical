var insertionSort = function (array) {
      const len = array.length;
      for (let i = 1; i < len; i++) {
          let now = array[i];
          let j = i - 1;
          while (j >= 0 && now < array[j]) {
              array[j + 1] = array[j];
              j--;
          }
          array[j + 1] = now;
      }
      return array;
  }
  // 空间复杂度O(1)
  // 时间复杂度O(n^2)
  // 稳定
